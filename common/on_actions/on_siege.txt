# country
# root = winning country, from = location
on_siege_won_country = {
	FROM = {
		if = {
			limit = { is_capital = no }
			cede_province = ROOT
			add_core = ROOT
		}
	}
}

# country
# root = losing country, from = location
on_siege_lost_country = {
	if = {
		limit = { FROM = { is_capital = yes } }
		every_war_enemy_country = { white_peace = ROOT }
	}
}
